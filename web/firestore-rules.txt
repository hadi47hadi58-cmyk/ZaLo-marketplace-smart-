rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users Collection rules
    match /users/{userId} {
      allow read, write: if true; // Allows clients and auth registration to save profiles safely
    }
    
    // Municipalities selection
    match /municipalities/{municipalityId} {
      allow read: if true;
      allow write: if true;
    }

    // Merchant Requests: client submits, admin reads/updates, client checks status
    match /merchant_requests/{requestId} {
      allow read, write: if true;
    }

    // Shops/Stores: public reading, merchants manage details
    match /shops/{shopId} {
      allow read, write: if true;
    }

    // Products: public reading, merchants manage items
    match /products/{productId} {
      allow read, write: if true;
    }

    // Orders: public reading/creation, merchants update status
    match /orders/{orderId} {
      allow read, write: if true;
    }

    // Messages / Live Chat: room participants read/write
    match /messages/{messageId} {
      allow read, write: if true;
    }
  }
}
