rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users Collection rules: only owner can read and write their profile data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Municipalities selection: public reading, write restriction to system managers
    match /municipalities/{municipalityId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.token.role == 'manager';
    }

    // Stores: public reading of approved stores, merchants can update their own store details
    match /stores/{storeId} {
      allow read: if resource.data.status == 'APPROVED' || (request.auth != null && request.auth.uid == resource.data.merchantId);
      allow create: if request.auth != null && request.auth.token.role == 'merchant';
      allow update: if request.auth != null && request.auth.uid == resource.data.merchantId;
      allow delete: if request.auth != null && request.auth.token.role == 'manager';
    }

    // Products: public reading, merchants manage items of their verified store
    match /products/{productId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.token.role == 'merchant';
    }

    // Orders: customers read/create own purchases, merchants update their store orders
    match /orders/{orderId} {
      allow read, create: if request.auth != null;
      allow update, delete: if request.auth != null && (
        request.auth.uid == resource.data.userId || 
        request.auth.uid == resource.data.merchantId ||
        request.auth.token.role == 'manager'
      );
    }

    // Messages / Live Chat: room participants read/write
    match /messages/{messageId} {
      allow read, write: if request.auth != null && (
        request.auth.uid == request.resource.data.senderId ||
        request.auth.uid == request.resource.data.receiverId
      );
    }
  }
}
